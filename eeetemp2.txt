A=imread('couple.bmp');
J1=mat2gray(A, [0 255]);
J2=mat2gray(A, [0 255]);
J3=mat2gray(A, [0 255]);

J4 = J1;
J5 = J2;

xmin=-0.5*0.2;
xmax=0.5*0.2;
n1=1;

for n=1:512
    for m = 1:512
        x=xmin+rand(1,n1)*(xmax-xmin);
        J1(n,m) = J1(n,m) + x;
    end
end

for n=1:512

end
J8 = J2;

for n=1:512

        x=xmin+rand(1,n1)*(xmax-xmin);
        J3(n,m) = J3(n,m) + x;
    end
end
J9 = J3;

M = 11
for n=1:512
    for m = 1:512
        k = 0;
        sum = 0;
        while k < (M-1)/2 
            if (m-k) > 0
                sum = sum + J1(n,m-k);
            end
            if m+k <= 512
                sum = sum + J1(n,m+k);                
            end
            k = k + 1;
        end
        sum = sum - J1(n,m);
        J1(n,m) = sum / M;
    end
end
subplot(3,3,1),imshow(J1);
